class User extends Jig { }


class UserGroup extends Jig {
    init() {
        this.users = []
    }

    addUser(user) {
        if (!this.users.includes(user)) {
            this.users.push(user)
        }
    }

    removeUser(user) {
        this.users = this.users.filter(u => user !== u)
    }
}
